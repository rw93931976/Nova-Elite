import { createClient } from '@supabase/supabase-js';

// Optional: use Gemini embedding (free tier generous) or OpenAI
async function generateEmbedding(text: string): Promise<number[] | null> {
  try {
    const res = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/text-embedding-004:embedContent?key=${import.meta.env.VITE_GEMINI_API_KEY}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          model: 'models/text-embedding-004',
          content: { parts: [{ text }] },
          taskType: 'RETRIEVAL_DOCUMENT'
        })
      }
    );
    const data = await res.json();
    return data.embedding?.values || null;
  } catch (err) {
    console.error('Embedding failed:', err);
    return null;
  }
}

public async storeMemory(input: string, response: string, context: any = {}) {
  const fullText = `${input}\n\nResponse: ${response}`;
  const summaryPrompt = `Summarize this interaction concisely (1-2 sentences), extract key facts, tags, and importance (0-1): ${fullText.slice(0, 4000)}`;
  
  // Get summary via LLM (reuse your callLLM helper if you have one)
  const summary = await this.callLLM(  // assuming you have this method
    "You are a memory summarizer. Be concise.",
    summaryPrompt,
    true
  ) || fullText.slice(0, 200);

  const embedding = await generateEmbedding(fullText);
  if (!embedding) return; // silent fail for now

  const { error } = await supabase
    .from('nova_memories')
    .insert({
      user_id: supabase.auth.getUser()?.id || null, // if using auth
      input_text: input,
      response_text: response,
      full_context: fullText,
      embedding,
      summary,
      tags: context.tags || [], // e.g. from your tag generator
      importance: 0.5, // could be LLM-scored later
      query_type: context.query_type || 'general'
    });

  if (error) console.error('Memory store failed:', error);
}
